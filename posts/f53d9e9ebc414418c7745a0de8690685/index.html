	from emp1 e inner join dept1 d on e.deptno = d.deptno
	group by e.deptno
	having count(*) > 3

	l.ename
	from emp1 e inner join emp1 l on e.mgr = l.empno
  • 得到平均工资大于2000的工作职种

	from emp1

  • 按部门统计员工数，员工人数越来越少排列，查出的第二名到第五名部门信息（列出部门名字，部门位置）
	select 
	d.*
	from emp1 e right join dept1 d on e.deptno = d.deptno
	group by e.deptno
	order by count(*) desc 
	limit 1,4
