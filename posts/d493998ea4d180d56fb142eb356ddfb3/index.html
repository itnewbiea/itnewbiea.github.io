    let obj2 = {};
    function clone(obj2, obj) {
      for (let key in obj) {
        let value = obj[key];
        console.log(value);
        if (Object.prototype.toString.call(value) === "[object Array]") {
          obj2[key] = [];
          clone(obj2[key], value);
        } else if (
          Object.prototype.toString.call(value) === "[object Object]"
        ) {
          obj2[key] = {};
          clone(obj2[key], value);
        } else {
          obj2[key] = value;
        }
      }
    }
    clone(obj2, obj);
