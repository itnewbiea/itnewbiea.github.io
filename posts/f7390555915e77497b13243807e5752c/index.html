【C++】对于结构体数组，如果没有初始化，那么全部数组内元素会被全部赋成0

#include <stdio.h>
struct{
	char *name;  //姓名
	int num;  //学号
	int age;  //年龄
	char group;  //所在小组
	float score;  //成绩
}Class[8] = {		//可以不给出数组长度，如果给的是空的，全部初始化成0
	{ "Li ping", 5, 18, 'C', 145.0 },
	{ "Zhang ping", 4, 19, 'A', 130.5 },
	{ "He fang", 1, 18, 'A', 148.5 },
	{ "Cheng ling", 2, 17, 'F', 139.0 },
	{ "Wang ming", 3, 17, 'B', 144.5 }
};
int main(){
	int i, num_140 = 0;
	float sum = 0;
	for (i = 0; i<8; i++){
		sum += Class[i].score;
		if (Class[i].score < 140) num_140++;		//判断语句，如果分数小于140  人数就 + 1
	}
	printf("sum=%.2f\naverage=%.2f\nnum_140=%d\n", Class[7].score, Class[7].score, Class[7].score);
	printf("sum=%.2f\naverage=%.2f\nnum_140=%d\n", sum, sum / 5, num_140);
	return 0;
}
