mysql添加ROW_NUMBER写法记录

select cs_id,user_id, count(1)as sl,
		ROW_NUMBER() OVER(ORDER BY cs_id asc, count(1) desc) as rownumber
		from sys_customer_mx
		where shenhe_state=1
		group by cs_id,user_id
		order by cs_id asc,sl desc
