编程求e=1+1/1!+1/2!+1/3!+...+1/n!，直到最后一项的值小于10的-5次方为止。

源代码：

#include <stdio.h>
int main()
{
	double e=0.0;
	int i=1,j=1,t=1;
	while(1.0/t >= 1e-5)
	{
		e=e+1.0/t;
		t=t*j;
		j++;
	}
	printf ("计算结果为：%lf\n",e);
	return 0;
}
