在方法内部获取方法的名称

StackTrace st = new StackTrace();

//0:取本身方法，1:取调用方法

StackFrame sf = st.GetFrame(0);

 

string methodName = sf.GetMethod().Name;

string className = sf.GetMethod().ReflectedType.Name;
