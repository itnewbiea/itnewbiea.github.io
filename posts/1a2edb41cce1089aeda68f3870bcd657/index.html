        const APP = {
            setup() {
                const open2 = () => {
                    ElMessage({
                        message: 'Congrats, this is a success message.',

                    })

                    ElMessage({
                        message: 'Im show.',
                        type: 'success',
                    })

                const value1 = ref('')


                const activeIndex = ref('1')
                const handleSelect = (key, keyPath) => {
                    console.log(key, keyPath)
                }

                return {
                    show,
                    open2,

                    activeIndex,
                    handleSelect,

                    value1
                }
            }
        }

        const app = createApp(APP)
        app.use(ElementPlus, {
            locale: ElementPlusLocaleZhCn,
        })
        app.mount('#app')
    </script>
