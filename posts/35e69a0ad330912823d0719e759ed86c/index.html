def clear_unnormal_data(df):
    #异常值处理,这里认为当连续的变量值在2.2个标准差外算作异常值，并删除异常值
    ####################数据##########
    #    col1  col2
    # 0     1    12
    # 1   200    17
    # 2     3    31
    # 3     5    53
    # 4     2    22
    # 5    12    32
    # 6    13    43
    ##################################
    ########## Begin ##########
    for i in df.columns:
        df_col=df[i]
        grade=(df_col - df_col,mean()) / df_col.std()
        df[i] = grade.abs() > 2.2
        df=df[df.sum(axis=1)==0]
    ########## End ##########
    return df
