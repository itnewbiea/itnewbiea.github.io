js数组去重 简单方法

 // 数组去重 方法一
        function sub3(arr) {
            let newArr = []
            arr.forEach(item => {
                if (newArr.indexOf(item) == -1) {
                    newArr.push(item)
                }
            });
            return newArr;
        }
        let ex1 = [1, 2, 3, 'r', 'r', 'e', 'e', 'd', 3, 1, 4, 6, 8]
        // 数组去重 方法二
        let ex2 = Array.from(new Set(ex1))
        console.log(ex2, sub3(ex1))
