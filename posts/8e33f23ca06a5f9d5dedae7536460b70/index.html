一、功能简介

本项目使用STM32F103C8T6单片机控制器，使用数码管模块、矩阵按键、无源蜂鸣器等。

主要功能：
系统运行后，蜂鸣器播放一首音乐，进入电子琴模式，
数码管默认显示0，可通过矩阵按键S1-S16控制1-7低中高音调，同时数码管显示对应音符。按键按下一直发音，松开停止。

void music_play2(void)
{
	u16 i=0;
	u16 length=0;
	
	length = sizeof(wind_rise)/sizeof(wind_rise[0]);
	
	for(i=0;i<(length/2);i++)
	{
		beep_on(wind_rise[i*2],260);
		delay_ms(wind_rise[i*2+1]*5);
	}
}

//系统数据设置
void sys_data_set(void)
{
	u8 key=0;
	u16 note[16]={L6,L7,M1,M2,M3,M4,M5,M6,M7,H1,H2,H3,H4,H5,H6,H7};
	
	key=KEY_Matrix_Scan();

	//有键按下
	if(key!=0)
	{
		sys_ctrl.note=key;
		beep_on(note[sys_ctrl.note-1],200);
	}
	else
	{
		sys_ctrl.note=0;
		beep_off();
	}
	//显示音符
	smg_display_num(sys_ctrl.note);
}


//应用控制系统
void app_sys_ctrl_demo(void)
{
	sys_parm_init();//系统参数初始化
	LED_Init();
	USART1_Init(115200);
	KEY_Matrix_Init();
	BEEP_Init();
	SMG_Init();
	music_play1();
	while(1)
	{
		sys_data_set();//系统数据设置
	}
}
