Arcpy批量裁剪（掩膜方法矢量裁剪栅格）（1个shp，多个raster文件）

 

# 当需要用一个shp文件裁剪多个raster情况，如：对需要广东省每天的MODIS数据时

数据准备：单个shp文件以及若干raster文件

语法

ExtractByMask (in_raster, in_mask_data)

# -*- coding: utf-8 -*-
 
import arcpy
from arcpy import env
from arcpy.sa import *

# Set environment settings
env.workspace = "D:/MODIS/M/13Q1"
outputpath="D:/MODIS/M/clip"

# Set local variables
rasters = arcpy.ListRasters("*", "tif")
inMaskData = "D:/Nercita/songxy/MODIS/M/shp/Guangdong.shp"

for raster in rasters:
    #去掉后缀和文件名中的'.'
    file_name = raster.split('.')[0]+raster.split('.')[1]+raster.split('.')[2]
    print(file_name)
    
    # Check out the ArcGIS Spatial Analyst extension license
    arcpy.CheckOutExtension("Spatial")

    # Execute ExtractByMask
    outExtractByMask = ExtractByMask(raster, inMaskData)
    
    # Save the output
    outExtractByMask.save(outputpath+'/'+file_name+'_clip.tif')

print('finish')
