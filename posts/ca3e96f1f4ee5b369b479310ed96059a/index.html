shell字段分隔符IFS，以逗号（，）为分隔符获取字符串内容

内部字段分隔符(Internal Field Separator, IFS)定义一个定界符，默认是空格。

 

#!/bin/bash
function test_for
{
   ifs_old=$IFS
   IFS=$','
   for i in $(echo "${1}")
   do
   echo "${i}"
   done
}

test_for "test1,test2"
