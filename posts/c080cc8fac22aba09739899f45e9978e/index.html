代码如下，调用数据集的output就可以

package wy

import org.apache.flink.api.common.io.OutputFormat
import org.apache.flink.api.common.operators.Order
import org.apache.flink.api.scala._
import org.apache.flink.configuration.Configuration
import org.apache.hadoop.hbase.{HBaseConfiguration, HConstants, TableName}
import org.apache.hadoop.hbase.client.{Connection, ConnectionFactory, Put}

object Phone {

  def main(args: Array[String]): Unit = {
    val exe = ExecutionEnvironment.getExecutionEnvironment
    val data: DataSet[String] = exe.readTextFile("D:\\IdeaObj\\flink\\src\\main\\resources\\phonenumber")
    val map = data.map(m => {
      val s = m.split(",")
      (s(0), s(1).toInt + s(2).toInt)
    })
    val sort: DataSet[(String, Int)] = map.sortPartition(1,Order.DESCENDING).setParallelism(1)
    sort.output(new Output())
    exe.execute()
  }
}

class Output extends OutputFormat[(String, Int)]{
  var con:Connection = null

  override def configure(configuration: Configuration) = {

  }

  override def close() = {
    con.close()
  }
  
  override def open(i: Int, i1: Int) = {
    val conf = HBaseConfiguration.create()
    conf.set(HConstants.ZOOKEEPER_QUORUM,"hdp1,hdp2,hdp3")
    conf.set(HConstants.ZOOKEEPER_CLIENT_PORT,"2181")
    con=ConnectionFactory.createConnection(conf)

  }


  override def writeRecord(it: (String, Int)) = {
    val table = con.getTable(TableName.valueOf("zy"))
    val put = new Put(it._1.getBytes())
    put.addColumn("f1".getBytes(),"liulian".getBytes(),it._2.toString.getBytes())
    table.put(put)

  }
}
