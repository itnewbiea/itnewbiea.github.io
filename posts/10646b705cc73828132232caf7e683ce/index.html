利用pom依赖上传jar包到nexus私服

<distributionManagement>
    <snapshotRepository>
        <id>nexus-snapshots</id>
        <name>admin</name>
        <url>http://url/repository/maven-snapshots/</url>
    </snapshotRepository>
    <repository>
        <id>nexus-releases</id>
        <name>admin</name>
        <url>http://url/repository/maven-releases/</url>
    </repository>
</distributionManagement>

Settings.xml

如果出现401， unauthorized 那肯定是你私服账号和命名配错了 请确认！！！
