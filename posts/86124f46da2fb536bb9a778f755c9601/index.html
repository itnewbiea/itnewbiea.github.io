complex complex::operator+(complex bl)
{
complex cl;
cl.a = a + bl.a;
cl.b = b + bl.b;
return cl;
}
complex complex::operator-(complex bl)
{
complex cl;
cl.a = a - bl.a;
cl.b = b - bl.b;
return cl;
}
complex complex::operator*(complex bl)
{
complex cl;
cl.a = a * bl.a - b * bl.b;
cl.b = b * bl.a + a * bl.b;
return cl;
}
complex complex::operator/(complex bl)
{
complex cl;
cl.a = (abl.a + b * bl.b) / (bl.abl.a - bl.bbl.b);
cl.b = (b * bl.a - a * bl.b) / (bl.abl.a - bl.b*bl.b);
return cl;
}

void mat::add()/加法/
{

if (m != l || n != k)
cout << “您输入的矩阵不能进行加法运算！” << endl;
else
{

for (j = 0;j < n;j++)
c[i][j] = a[i][j] + b[i][j];
for (i = 0;i < m;i++)
{
cout << endl;
for (j = 0;j < n;j++)
cout << c[i][j] << " ";
}
}
}
