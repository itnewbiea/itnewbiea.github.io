将本地jar包上传到私服

### 本地jar上传本地maven
mvn install:install-file -Dfile=E:/IdeaProjects/kingbase8-8.6.0.jar -DgroupId=com.kingbase8 -DartifactId=kingbase8 -Dversion=8.6.0 -Dpackaging=jar

### 本地jar上传私服
mvn deploy:deploy-file -Dmaven.test.skip=true -DgroupId=com.kingbase8 -DartifactId=kingbase8 -Dversion=8.6.0 -Dpackaging=jar -Dfile=E:/IdeaProjects/kingbase8-8.6.0.jar -Durl=http://192.168.1.70:8081/repository/maven-releases/ -DrepositoryId=nexus-releases
