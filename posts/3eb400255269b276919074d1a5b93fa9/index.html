elementui表格插槽用法

如下:

<el-table-column prop="approveStatus" width="100px" label="发布状态">
          <template slot-scope="scope">
          <span>{{getDictName(scope.row.approveStatus, "approveStatus")}}</span>
          </template>
          <template slot-scope="scope">
          <span v-if="scope.row.approveStatus == '0'">待发布</span>
          <span v-if="scope.row.approveStatus == '1'">已发布</span>
          <span v-if="scope.row.approveStatus == '2'">发布审批中</span>
          <span v-if="scope.row.approveStatus == '3'">发布审批不通过</span>
          <span v-if="scope.row.approveStatus == '4'">已下架</span>
          <span v-if="scope.row.approveStatus == '5'">下架审批中</span>
          <span v-if="scope.row.approveStatus == '6'">下架审批不通过</span>
        </template>
          </el-table-column>
