top.layer.open 子页面按钮点击关闭刷新父页面

// 父页面打开弹框
top.layer.open({
			......
})

// 子页面方法
$("#button").click(function(){
	var index = parent.layer.getFrameIndex(window.name);
	parent.layer.close(index);// 关闭弹出层
	parent.window.location.reload(); // 刷新父页面
})
