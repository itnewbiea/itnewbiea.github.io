element-ui 改变单选框，多选框的选中颜色

@element-ui 改变单选框，多选框的颜色
注意 是写在less文件中，并在vue.config.js中配置

@ztColor: #da222a;
// 全局单选多选按钮，选中颜色改变
/deep/ .is-checked {
    .el-checkbox__inner {
        background-color: @ztColor;
        border: 1px solid @ztColor;
    }
    .el-checkbox__label {
        color: @ztColor;
    }
}
/deep/ .el-radio{
    &.is-checked {
        &.is-bordered {
            border: 1px solid @ztColor;
        }
        .el-radio__label {
            color: @ztColor;
        }
        .el-radio__inner {
            background-color: @ztColor;
            border: 1px solid @ztColor;
        }
    }
}
