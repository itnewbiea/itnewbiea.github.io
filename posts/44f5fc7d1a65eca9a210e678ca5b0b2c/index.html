求质数 js

    <script>
            var a = 57;
            var flag = true;
            for (var i = 2; i < a; i++) {
                if (a % i == 0) {
                    flag = false;
                    break;
                }
            }
            if (flag) {
                console.log(a + '是质数');
            } else {
                console.log(a + '不是质数');
            }
        </script>
