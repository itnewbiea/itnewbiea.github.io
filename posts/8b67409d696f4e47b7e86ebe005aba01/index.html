解决方案其实很简单，如下：

//这里unlockNumber可能为空
BigDecimal unlockNumber = unlockAccountInfo.getUnlockNumber();
if (unlockNumber == null)
{
	//错误提示，或者进行其它操作
	throw new BusinessException(CommonErrorCode.UNLOCK_NULL);
}
