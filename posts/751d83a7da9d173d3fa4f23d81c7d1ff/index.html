实现自动批量修改文件名

import os
root='./images' #文件目录
files=os.listdir(root)
file=os.path.splitext(files[0])
filename, type=file
for i in range(len(files)):
  os.rename(files[i], str[i]+type) #str[i]为新名字，可以改为其他的
