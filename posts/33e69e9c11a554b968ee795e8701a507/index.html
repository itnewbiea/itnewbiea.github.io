Leetcode2467-树上最大得分和路径

两遍dfs：

        第一遍dfs把所有结点的父节点跑出来，然后预处理bob回到根结点的路径上的amount值。

        第二遍dfs把所有子节点的得分算出来取最大值即可。

class Solution {
public:
    vector<vector<int>> e;
    vector<int> fa;
    vector<int> pre;
    int ans=-10000000;
    int n;
    void dfs1(int x,int p){
        fa[x]=p;
        for(auto u:e[x]){
            if(u==p) continue;
            dfs1(u,x);
        }
    }
    void dfs2(int x,int cur,vector<int>& amount){
        cur+=amount[x];
        if(e[x].size()==1&&e[x][0]==fa[x]){
            ans=max(ans,cur);
        }
        for(auto ed:e[x]){
            if(ed!=fa[x]) dfs2(ed,cur,amount);
        }
    }
    int mostProfitablePath(vector<vector<int>>& edges, int bob, vector<int>& amount) {
        n=edges.size()+1;
        e.resize(n);
        fa.resize(n);
        for(auto p:edges){
            e[p[0]].push_back(p[1]);
            e[p[1]].push_back(p[0]);
        }
        dfs1(0,-1);
        for(;~bob;bob=fa[bob]){
            pre.push_back(bob);
        }
        for(int i=0;i<pre.size()/2;i++){
            amount[pre[i]]=0;
        }
        if(pre.size()&1) amount[pre[pre.size()/2]]/=2;
        dfs2(0,0,amount);
        return ans;
    }
};

时间复杂度：O（n）

空间复杂度：O（n）
