多个confirm弹窗

async openHTML() {

        let arr = ['1', '2', '3']

       

        for await (const item of arr) {

          console.log(item)

              this.$confirm(item, '提示', {

          confirmButtonText: '确定',

          cancelButtonText: '取消',

          type: 'warning'

        }).then(() => {

          this.$message({

            type: 'success',

            message: '删除成功!'

          });

        }).catch(() => {

         

          this.$message({

            type: 'info',

            message: '已取消删除'

          });          

        });

        }

这种情况下    不能控制弹窗取消后  后面的弹窗不显示 如果不需要这部分功能还可以这样写。

否则 ，用递归写法：

abc(arr){

this.count++

if(this.count<arr.length){

this.$confirm(arr[this.count], '提示', {

          confirmButtonText: '确定',

          cancelButtonText: '取消',

          type: 'warning'

        }).then(() => {

         this.abc(arr)

        }).catch(() => {

         this.count = -1

          this.$message({

            type: 'info',

            message: '已取消'

          });          

        });

}

}
