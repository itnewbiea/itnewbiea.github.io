<!DOCTYPE html>
<html>
	<head>
		<!--引入jquery，jquery.qrcode-->

		<script type="text/javascript"src="http://cdn.staticfile.org/jquery.qrcode/1.0/jquery.qrcode.min.js"></script>

	</body>

		});
	</script>
</html>
