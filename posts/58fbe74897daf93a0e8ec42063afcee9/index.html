golang 控制方法的访问时间

package main

import (
	"time"
	"fmt"
)

func main()  {
	limitTime(funcA, 6*time.Second)
}

func limitTime(f func(),timer time.Duration)  {
	ok := make(chan struct{})
	go func() {
		f()
		ok<- struct{}{}
	}()
	select {
	case <- time.After(timer):
		fmt.Println("time out")
	case <-ok:
	}
}

func funcA()  {
	time.Sleep(time.Second*10)
	fmt.Println("there is a function")
}
