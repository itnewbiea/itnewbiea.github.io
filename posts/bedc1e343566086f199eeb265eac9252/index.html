        if(!$ManagerCheck->check($params)) {
            $res = [
                "code"=>201,
                "message"=>"账号密码格式不正确",
            ];
            return json($res);

        }
        if($result) {
            $res = [
                "code"=>202,
                "message"=>"账号已经存在",
            ];
            return json($res);

        }
        $data = [
            'username' => $params['username'],
            'password' => password_hash($params['password'],PASSWORD_DEFAULT),
            'create_at' => date('Y-m-d H:i:s'),
            'update_at' => date('Y-m-d H:i:s'),
            'last_ip' => Request::ip()
        ];
        Db::name("admin")->insert($data);
        $res = [
            "code"=>200,
            "message"=>"账号创建成功",
            "data"=>$data
        ];
