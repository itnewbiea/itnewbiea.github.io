MATLAB: 替换底色

%% 替换底色
% 读入证件照图像
img = imread("C:\Users\hqh\Desktop\屏幕截图 2023-05-30 233907.png");
% 将图像转换为RGB格式
img = im2double(img);
if size(img,3) == 1
    img = repmat(img,[1,1,3]);
end
% 计算图像的灰度图像
gray_img = rgb2gray(img);
% 通过Otsu阈值法得到一个二值化的掩膜mask，用于分割出人像和背景
level = graythresh(gray_img);
mask= imbinarize(gray_img, level);
figure
imshow(mask) %人脸+背景为1
