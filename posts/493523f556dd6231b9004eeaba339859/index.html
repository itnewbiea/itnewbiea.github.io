类型四：（任意输入打印的行数）

代码：

int main()
{
	int i,j,k,n;
	printf("请输入想要打印的行数：");
	scanf("%d",&n);
	for(i=1;i<=n;i++)//控行数 
	{
		for(k=n-1;k>i-1;k--)//控空格数 
		printf(" ");
		for(j=1;j<=2*i-1;j++)//控*数

 return 0;
