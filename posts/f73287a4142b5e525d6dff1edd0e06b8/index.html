java ID最后一位校验位生成规则

        /**
	 * @author SevenFin
	 * @date 2017-11-10 上午10:52:08
	 * @param id ID前17位
	 * @return ID第18位 校验位
	 */
	public static String verifyId(String id) {
		int count = 0;
		char[] charArr = id.toCharArray();
		for (int i = 0; i < charArr.length; i++) {

			int n = Integer.parseInt(charArr[i] + "");
			count += n * (Math.pow(2, 17 - i) % 11);
		}
		
		switch (count % 11) {
		case 0: return "1";
		case 1: return "0";
		case 2: return "X";
		default:
			return 12 - (count % 11) + "";
		}
		
	}
