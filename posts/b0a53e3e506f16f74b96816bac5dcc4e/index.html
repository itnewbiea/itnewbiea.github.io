Ulimit配置

vi /etc/security/limits.conf

* hard nofile 655350
* soft nofile 655350
* hard memlock unlimited
* soft memlock unlimited
* soft  core   unlimit
* hard  core   unlimit
* soft  fsize  unlimited
* hard  fsize  unlimited
* soft  data   unlimited
* hard  data   unlimited
* soft  nproc  65535
* hard  nproc  63535
* soft  stack  unlimited
* hard  stack  unlimited

或者：
*               -       nofile         100000
*               -       nproc          100000

无限制配置：

# vim /etc/security/limits.conf
	*               -       nofile         100000000
	*               -       memlock        unlimited
	*               -       core           unlimited
	*               -       fsize          unlimited
	*               -       data           unlimited
	*               -       nproc          unlimited
	*               -       stack          unlimited
	root - as unlimited


# vim /etc/sysctl.conf
	kernel.pid_max=100000
	net.core.somaxconn = 32768   # socket的最大客户端等待队列, 默认是128.
	vm.swappiness=0      #最大限度使用物理内存，然后才是 swap空间
	net.ipv4.ip_local_port_range = 30100 50000 #动态端口范围
	kernel.shmmax = 17179869184   #此参数配置不小于物理内存的1/4，单位B
	vm.overcommit_memory=0
	fs.nr_open=100000000  #限制 nofile 最大值
