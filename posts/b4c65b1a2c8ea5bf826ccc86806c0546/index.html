JS

var pageObj = {
  data: {
    isChecked1: false,
    isChecked2: true,
    isChecked3: false,
    isChecked4: true,
    isChecked5: false,
    isChecked6: true
  }
};

for (var i = 0; i < 7; ++i) {
  (function (i) {
    pageObj['changeSwitch' + i] = function (e) {
      var changedData = {};
      changedData['isChecked' + i] = !this.data['isChecked' + i];
      this.setData(changedData);
    }
  })(i)
}
Page(pageObj);

###switch组件的属性

  1. checked：是否选中----false、true
  2. type：开关选择器的样式----switch, checkbox
  3. color：switch 的颜色，同 css 的 color
