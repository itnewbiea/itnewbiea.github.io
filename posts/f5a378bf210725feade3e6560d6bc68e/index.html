docker-compose部署elasticsearch-exporter和kafka-exporter

version: '3'
services: 
  elasticsearch-exporter:
    image: bitnami/elasticsearch-exporter:latest
    commmand:
      - '--es.uri:http://elasticsearch:9200'
    restart: always
    ports:
      - "9114:9114"
 
  kafka-exporter:
    image: bitnami/kafka-exporter:latest
    command:
      - '--kafka.server=192.168.1.36:9092'
      - '--kafka.server=192.168.1.36:9092'
    restart: always
    port:
      - "9308:9308"
