初始化私有变量：

private string[] red = { "01","02","03","04","05","06",
        "07","08","09","10","11","12","13","14","15","16","17"};
        private string[] blue = {"01","02","03","04","05","06",
        "07","08","09","10" };
        private bool begin = true;

开始按钮：

private void button1_Click(object sender, EventArgs e)
        {
            button1.Text = "已经开始";
            button1.Enabled = false;
            button2.Enabled = true;
            begin = true;

            TaskFactory f = new TaskFactory();
            List<Task> list = new List<Task>();
            foreach(var item in groupBox1.Controls)
            {
                if(item is Label)
                {
                    ((Label)item).Text = "00";
                }
            }
            foreach( var item in groupBox1.Controls)
            {
                if(item is Label)
                {
                    Label lbl = item as Label;
                    list.Add(f.StartNew(() =>
                    {
                        Random r = new Random();
                        while (begin)
                        {
                            string txt = "";
                            if (lbl.Name.Contains("Red"))
                            {
                                lock (obj)
                                {
                                    
                                    txt = red[r.Next(0, 17)];
                                    Update(lbl, txt);
                                    
                                }
                            }
                            else
                            {
                                txt = blue[r.Next(0, 10)];
                                Update(lbl, txt);
                            }
                            Thread.Sleep(10);
                        }
                    }));
                }
            }
            f.ContinueWhenAll(list.ToArray(), tList => this.ShowResult());
        }

数字更新：

private void Update(Label label , string text)
        {
            this.Invoke(new Action(() =>
            {
                label.Text = text;
            }));
        }

结束按钮：

private void button2_Click(object sender, EventArgs e)
        {
            button1.Text = "开始";
            button1.Enabled = true;
            button2.Enabled = false;
            begin = false;
        }
