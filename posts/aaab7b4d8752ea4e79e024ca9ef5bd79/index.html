输入一个十进制数 N，将它转换成 R 进制数输出。

输入一个十进制数 N，将它转换成 R 进制数输出。 要求： 为每个测试实例输出转换后的数，每个输出占一行。如果 R 大于 10，则对应 的数字规则参考 16 进制（比如，10 用 A 表示，等等）。 注：以输入 0 0 代表结束 输入样例: 7 2 23 12 -4 3 0 0 输出: 111 1B -11 

#include"stdio.h"
#include"stdlib.h"
int main()
{
    int num[50],base[50],r[50];
    char n[128];
    int i = 0,j = 0,a;
    for(i=0;i>=0;i++)
    {
        scanf("%d",&num[i]);//输入任意的十进制
        scanf("%d",&base[i]);//输入想要转的进制
        if((num[i]==0)&&(base[i]==0))
            break;
    }
    a=i;
    for(i=0;i<a;i++)
    {
        r[i]=num[i];
        while(num[i] != 0)
        {
            if(num[i]<0)
                num[i]=-num[i];//取绝对值
            n[j] = num[i] % base[i];//取余
            num[i] = num[i] / base[i];
            j++;
        }
        if(r[i]<0)
            printf("-");
        for(j--;j>=0;j--)
        {
            if(n[j]>=10)
                printf("%c",n[j]-10+'A');
            else
                printf("%d",n[j]);
        }
        j=0;
        printf("\n");
    }
    return 0;
}
