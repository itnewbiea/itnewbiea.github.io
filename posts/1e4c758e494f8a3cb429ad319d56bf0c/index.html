from math import e
def func(x):
    return x-(x*pow(e,x)-1)/(pow(e,x)*(1+x))
x=0.5
print(f"k={0:02d},{x:.10f}")
for i in range(10):
    x=func(x)
    print(f"k={i+1:02d},{x:.10f}")

简化牛顿法求解 x e x − 1 = 0 xe^x-1=0 xex−1=0

from math import e
def func(x):
    return x*pow(e,x)-1
x=0.5
print(f"k={0:02d},{x:.10f}")
C=1/(pow(e,x)*(1+x))
for i in range(10):
    x=x-C*func(x)
    print(f"k={i+1:02d},{x:.10f}")
