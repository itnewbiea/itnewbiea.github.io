server.on('request', async function (req, res) {

  const {
    method,
    url,
  } = req

  if (method === 'GET') {
    if (url === '/user') {
      // 查找集合中的数据（user)
      const data = await Model.find();
      res.end(JSON.stringify({
        code: 0,
        data: {
          list: data || []
        }
      }))
    }
  } else if (method === 'DELETE') {
    if (url === '/user') {
      // 查找集合中的数据（user)
      const data = await Model.find({
        name: '木子'
      });
      const idList = data.map(v => v._id)
      const list = await Model.deleteMany({
        _id: {
          $in: idList || []
        }
      });
      res.end(JSON.stringify({
        code: 0,
        data: {
          list: data || [],
          lo: list
        }
      }))
    }
  }



})
