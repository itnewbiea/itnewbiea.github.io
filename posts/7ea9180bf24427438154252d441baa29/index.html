matlab练习程序（灰度拉伸）

cl;
img=imread('TestDatabase/1.bmp');
imshow(img);
img=double(img);
figure,imshow(mat2gray(img))
ma=double(max(max(img)));
mi=double(min(min(img)));
img=(255/(ma-mi))*img-(255*mi)/(ma-mi);
img=uint8(img);
figure,imshow(img)
