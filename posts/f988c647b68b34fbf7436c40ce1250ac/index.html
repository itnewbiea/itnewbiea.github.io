4.3.8break和continue

break:中断,中断当前的循环结构(switch中是中断当前的switch结构)
continue:继续,中断当前继续下次

//	break和continue
public static void main(String[] args) {
    for(int i = 1;i<=10;i++) {
        if(i == 7) {
            // 只要在循环之内,就是中断当前的循环
            break;
        }
        System.out.println(i);
    }
    System.out.println("--------------------------------------");
    for(int i = 1;i<=10;i++) {
        if(i == 7) {
            // 中断当前,继续下次
            continue;
        }
        System.out.println(i);
    }
}
