src/store/modules/haha.js

const haha = {
 state: {
    count: 100,
    text: "学习Vue.js 3管理后台开发"
},
 getters: {
    getCount: state => {
        return state.count + 10; 
    },
    getText: state => {
        return "我正在努力" + state.text
    }
},
 mutations: {
    SET_COUNT(state)  {
        state.count = 10;
    },
    SET_TEXT(state, payload)  {
        state.text = payload;
}
},
actions: {}


}

export default haha
