js获取对象数组中的id集合

const ids = Array.from(this.checkList,({ id }) => id);
// ["16921C72E0C64002","16898B37FC66E002"]

const ids = this.checkList.map(item => Object.assign({}, {'id': item.id}));
// [{"id":"16921C72E0C64002"},{"id":"16898B37FC66E002"}]
