pyecharts生成柱形图

1.版本：pyecharts版本1.9.0

2.功能：pandas数据框生成柱形图

from pyecharts.charts import Bar
from pyecharts import options as opts
import pandas as pd

a = {'num':[1,2,3,4,5],'name':['a','bb','c','d','f']}
df = pd.DataFrame(a)
mean_kh = df['name'].values.tolist()
label = df['num'].values.tolist()
bar=(
    Bar()
    .add_xaxis(mean_kh)
    .add_yaxis('平均价位',label)
    .reversal_axis()   #翻转
    .set_series_opts(label_opts=opts.LabelOpts(position="right"))
    .set_global_opts(title_opts=opts.TitleOpts(title="测试"))
)
bar.render("aa.html")
