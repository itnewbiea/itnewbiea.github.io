数据更新,记录数据更新情况,数据操作前后变化情况

--原始数据表

create table T(姓名 char(4),外号 char(4))
insert into T
select '张三','老三'
union all
select '李四','老四'

--数据变化情况表结构

create table T_BAK(原值 char(4),新值 char(4))

--创建触发器

Create trigger T_Update
on T
after update
as
begin
  insert into T_BAK
  select a.姓名,b.姓名 from deleted a,inserted b where a.外号 = b.外号
end

--更新

update T set 姓名 = '2'

--查看结果

select * from T_BAK

--结果

原值   新值  
---- ----
张三   2  
李四   2  

（所影响的行数为 2 行）
