1.什么是三子棋

  • 三子棋是一种民间传统游戏，又叫九宫棋、圈圈叉叉棋、一条龙、井字棋等。游戏分为双方对战，双方依次在9宫格棋盘上摆放棋子，率先将自己的三个棋子走成一条线就视为胜利，而对方就算输了，但是三子棋在很多时候会出现和棋的局面。

2.三子棋思想

  • 首先我们需要创建一个9*9的正方形框架
  • 然后由玩家和电脑依次在小型方格内下入他们的棋，需要判断这个格是否为空，如果为空则可以在格内下棋，否则需要从新寻找空格
  • 如果一个人三个棋在同一行或同一列或同一斜线上，则判断获胜.

5.game.h（头文件）

6.game.c（游戏部分）

这部分我们要实现棋盘的初始化，棋盘的打印，玩家下棋，电脑下棋，判断输赢以及棋盘是否满了。

6.1棋盘初始化

打印9*9的正方形，里面是9个空格
row和col是棋盘和行和列

void init_board(char board[ROW][COL], int row, int col)
{
	int i = 0;
	int j = 0;
	for (i =0; i < row; i++)
	{
		for (j = 0; j < col; j++)
		{
			board[i][j] = ' ';
		}
	}
}

6.2打印棋盘

这部分是打印棋盘，实际上就是给初始化的棋盘空格，加上横线和竖线

棋盘的搭建的本质是留出一片空间，玩家或者电脑在棋盘上进行操作。每一行和每一列都要设置分割线进行区分，才能构建完成打印出的棋盘大概就是一个“井”框架。

void display_board(char board[ROW][COL], int row, int col)
{
	int i = 0;
	
	for (i = 0; i < row; i++)
	{
		int j = 0;
		for (j = 0; j < col; j++)
		{
			printf(" %c ",board[i][j]);
			if (j < col - 1)
			{
				printf("|");
			}
		}
		printf("\n");
		if (i < col - 1)
		{
			for (j = 0; j < col; j++)
			{
				printf("---");
				if (j < col - 1)
			    {
				  printf("|");
			    }
			}
			printf("\n");
			
		}
	}

}

6.3玩家下棋

实际上就是判断这个格是否为空，为空的话就填入*，否则就重新选择空格。

void player_move(char board[ROW][COL], int row, int col)
{
	int x = 0;
	int y = 0;
	printf("玩家下棋>\n");
	while (1)
	{
		printf("请输入要下棋的坐标:>");
		scanf_s("%d %d", &x, &y);
		if (x >= 1 && x <= row && y >= 1 && y <= col)
		{
			if (board[x - 1][y - 1] == ' ')
			{
				board[x - 1][y - 1] = '*';
				break;
			}
			else
			{
				printf("坐标被占用，重新选择：\n");
			}
		}
		else
		{
			printf("坐标非法，重新输入\n");
		}
	}
}

6.4电脑下棋

这部分和玩家一样，实际上就是判断这个格是否为空，为空的话就填入#，否则就重新选择空格。

不过不同是电脑是通过产生随机数当做棋盘的位置，
rand()%row 就是产生1–row之间的数

void computer_move(char board[ROW][COL], int row, int col)
{
	
	printf("电脑下棋>\n");
	while (1)
	{
		int x = rand()%row;
		int y = rand()%col;
		if (board[x ][y ]==' ')
		{
			board[x][y] = '#';
			break;
		}

	}
}

6.5判断是否赢了

就是判断棋盘是否有行，列，斜线，三个棋都一样的

如果返回的是（*）那么玩家赢，如果返回的是（#）那么电脑赢，都没赢的话判断棋盘是否满了
满了的话，则为平局，，否则继续。

char is_win(char board[ROW][COL], int row, int col)
{
	int i = 0;
	for (i = 0; i < row; i++)
	{
		if (board[i][0] == board[i][1] && board[i][1] == board[i][2] && board[i][0] != ' ')
		{
			
			return board[i][0];
		}
	}
	for (i = 0; i < col; i++)
	{
		if (board[0][i] == board[1][i] && board[1][i] == board[2][i] && board[0][i] != ' ')
		{

			return board[0][i];
		}
		if (board[0][0] == board[1][1] && board[1][1] == board[2][2] && board[0][0] != ' ')
		{

			return board[1][1];
		}
		if (board[0][2] == board[1][1] && board[1][1] == board[2][0] && board[1][1] != ' ')
		{
			return board[1][1];
		}
		if (is_full(board, row, col) == 1)
		{
			return 'Q';
		}

		//继续
		return 'C';
	}
}

6.6 判断棋盘是否为空

判断棋盘是否有空格，还有空格就还可以下棋，满了的话返回1再返回Q，视为平局

static int is_full(char board[ROW][COL], int row, int col)
{
	int i = 0;
	for (i = 0; i < row; i++)
	{
		int j = 0;
		for (j = 0; j < col; j++)
		{
			if (' ' == board[i][j])
			{
				return 0;
			}
		}
	}

	return 1;
}
