js写登录界面 回车直接登录

在js中写入以下代码，并在点击事件上绑定

function keyLogin(){
  if (event.keyCode==13) {//回车键的键值为13
      var toClick=document.getElementById("keyBtn")
         toClick.click();  //调用登录按钮的登录事件
}
}`
function IbtnEnter_onclick() {   
            keyLogin();  
        }

在jsp文件中写页面

<div class="login" onkeydown="keyLogin()">
    <div class="body">
        <input class="userInputUserName" placeholder="请输入用户名""/>
        <input class="userInputPassword" placeholder="请输入密码" "/>
        <button id="keyBtn"class="btn"onclick="IbtnEnter_onclick()">           登录</button>
    </div>
</div>
