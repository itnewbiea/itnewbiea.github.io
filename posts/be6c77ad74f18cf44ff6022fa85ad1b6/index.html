let match = false;

const findMatchRecursion = (activity, activityItems) => {

for (let i = 0; i < activityItems.length; i += 1) {

if (activityItems[i].activity === activity) {

match = true;

break;

}

if (activityItems[i].items) {

findMatchRecursion(activity, activityItems[i].items);

}

}

return match;

};
