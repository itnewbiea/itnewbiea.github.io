     HAR2EV = 27.211386246
     KCALMOL2EV = 0.04336414
     
     conversion = torch.tensor([
         1., 1., HAR2EV, HAR2EV, HAR2EV, 1., HAR2EV, HAR2EV, HAR2EV, HAR2EV, HAR2EV,
         1., KCALMOL2EV, KCALMOL2EV, KCALMOL2EV, KCALMOL2EV, 1., 1., 1.
     ])
