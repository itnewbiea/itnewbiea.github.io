C语言 怎么输入3个数 输出最大值

键入三个数 输出最大值问题 

#include<stdio.h>
int main(){
	int a,b,c;
    printf("请输入三个整数:\n");
	scanf("%d %d %d",&a,&b,&c);
	if (a>b&&a>c)
	printf("%d",a);
	else if(b>a&&b>c)
	printf("%d",b);
	else
	printf("最大值为 %d",c); 
    return 0;
}
