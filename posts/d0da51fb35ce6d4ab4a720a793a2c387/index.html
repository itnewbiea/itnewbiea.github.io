python 文件查找

一、获取文件信息

代码如下（示例）：

import os
path = r'C:\Windows\System32\calc.exe'
print(os.path.dirname(path))    #获取目录名称
print(os.path.basename(path))   #获取基本名称

二、查找文件

代码如下（示例）：

#####  查找文件
import os
import argparse

def search_file(dir_path, filters) :
    print("search begin！！！！")
#     print("search files begin in dir {}".format(dir_path))
    for cur_dir, sub_dir, files in os.walk(dir_path) :
        for file in files :
            if os.path.splitext(file)[1] in filters :
                file_abs_path = os.path.join( cur_dir, file )
                print( file_abs_path[len(dir_path):] )
    print("search end！！！！")
    
def search_namefile( a , b ):
    for root,dirs,files in os.walk(a,topdown=True):
        if b in files:
            print('找到了')
            print(root)
            break
        else:
            pass
        
if __name__ == '__main__' :
    s = int(input('【根据文件后缀名查询文件输入1，根据文件名查询文件输入2】：'))
    if s == 1:
        a = input('请输入想要查找的磁盘或者请输入想要查找的路径：')
        b = input('请输入想要查文件文件后缀名：')
        search_file( a , [ b ] )
    elif s == 2:
        a=input('请输入想要查找的磁盘：')+':/'
