C语言程序,利用公式 e=1 + 1/1! + 1/2! + 1/3! + ... + 1/n! 求出e的近似值,其中n由用户输入。

#include <stdio.h>
void main(){
    double e=1,temp=1;
    int i,n;
    scanf("%d",&n);
    for(i=1;i<=n;i++){

        temp*=i;//进行阶乘操作
        e+=1/(double)temp;


    }
    printf("%f",e);

    }
