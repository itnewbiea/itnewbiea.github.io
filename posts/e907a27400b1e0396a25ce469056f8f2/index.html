	<a-form-item
        label="身份证号"
        :labelCol="{span: 5}"
        :wrapperCol="{span: 19}"
      >
        <a-input
          v-decorator.trim="['idCard',validatorRules.idCard]"
          placeholder="请输入身份证号"
        />
    </a-form-item>
</a-form>

data(){
	return{
		form:this.$form.createForm(this),
