JS中Key-Value操作


-- 声明对象
var map = {}

-- 添加/修改键值对
map['key1'] = 'key001'
map['key2'] = 'key002'

console.log(map['key1']) -- 输出：key001
console.log(map['key2']) -- 输出：key002

-- 遍历对象
for(var t in map){
    if(map.hasOwnProperty(t)){
        console.log(t + ' = ' + map[t]);
    }
}

-- 删除键值对
delete map['key1']
