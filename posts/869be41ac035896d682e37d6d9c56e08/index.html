function result=reduce(A)
global E;
if length(unique(A(:,2)))==1
    E=[E;A(1,:)];
    %result=B;
else
    reduce(A(1:length(A)/4,:))
    reduce(A(length(A)/4+1:length(A)/2,:))
    reduce(A(length(A)/2+1:length(A)*3/4,:))
    reduce(A(length(A)*3/4+1:length(A),:))
end
result=E;
