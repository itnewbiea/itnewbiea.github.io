C语言简单的爱心代码

                前两天看到网上有很多人在找爱心代码，小编立刻就坐不了，就用C语言简单的写了一个，希望对大家能有帮助。

#include<stdio.h>
#include<Windows.h>

int main()
{
    for(int i=0;i<2;i++)
    {
        float x,y,a;
     for(y=1.5; y>-1.5; y-=0.1)
     {
          for(x=-1.5; x<1.5; x+=0.05)
          {
               a=x*x+y*y-1;
               putchar(a*a*a-x*x*y*y*y<=0.0?'*':' ');
          }
          system("color 0c");
          putchar('\n');
     }
    }
    return 0;
}
