字符串加密-后移4位

#include<stdio.h>
int main()
{

   char c;
   
   while((c=getchar())!='\n')
   {
   	if(c>='a'&&c<='z'||c>='A'&&c<='Z')
   	{
   		c=c+4;
   		
   		if(c>'z'||c>'Z'&&c<='Z'+4)
   			c=c-26;
   		
	   }
	   printf("%c",c);
   }

	return 0;
}
