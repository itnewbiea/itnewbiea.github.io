//查找，最后一次出现x
#include <stdio.h>
int main()
{
    int a[10],i,x,k=0;
    for(i=0;i<=9;i++)
      scanf("%d",&a[i]);
    scanf("%d",&x);
    for(i=9;i>=0;i--)
    {
        if(a[i]==x)
        {
            printf("%d在该数组中最后一次出现的位置：%d\n",x,i);
            k++;
            break;
        }
    }
    if(k==0)printf("NotFound\n");
}
