how_xin=10

color_red="\e[1;31m"
color_null="\e[0m"
color_purple="\e[0;35m"
clear
for i in {1..100}
do

echo -e "$color_red     LoveLo    LoveLo    $color_null"
echo -e "$color_red LoveLoLoveLoLoveLoLoveLo$color_null"
echo -e "$color_red  LoveLoLoveLoLoveLoLove $color_null"
echo -e "$color_red   LoveLoLoveLoLoveLoLo  $color_null"
echo -e "$color_red     LoveLoLoveLoLove    $color_null"
echo -e "$color_red       LoveLoLoveLo      $color_null"
echo -e "$color_red          LoveLo         $color_null"
echo -e "$color_red            V           $color_null"
sleep 0.5
clear
echo -e "$color_purple     LoveLo    LoveLo    $color_null"
echo -e "$color_purple LoveLoLoveLoLoveLoLoveLo$color_null"
echo -e "$color_purple  LoveLoLoveLoLoveLoLove $color_null"
echo -e "$color_purple   LoveLoLoveLoLoveLoLo  $color_null"
echo -e "$color_purple     LoveLoLoveLoLove    $color_null"
echo -e "$color_purple       LoveLoLoveLo      $color_null"
echo -e "$color_purple          LoveLo         $color_null"
echo -e "$color_purple            V           $color_null"
sleep 0.5
clear
done
