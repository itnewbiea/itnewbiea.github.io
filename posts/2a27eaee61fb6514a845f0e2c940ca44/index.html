layui.table.reload 会携带上次的where 参数解决

table.render({ 
	id: "tableId" ,
	...... ,
	done: function() { 
		this.where = {} //本次加载完成直接置空where搜素条件!!!!!!!
	} 
}); 
//重载的时候重新携带参数
table.reload('tableId',{ 
	where: data,//表格重载时设置新的搜索参数 
});
