js 根据id取出数组中的对象

var  arr=[
    {
        id: 1,
        name: 'a'
    },
    {
        id: 2,
        name: 'b'
    },
]

var obj=arr.find(function (x) {
    return x.id === 1
})
console.log(obj);//打印出来是{id:1,name:a }，之后根据自己的需求点出来就可以了。
