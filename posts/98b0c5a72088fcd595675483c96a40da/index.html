MVC ajax

 var options = {
                        url: "@Url.Action("Save")",      //controller名称
                        type: "POST",
                        data: $("#Form").serialize(),    //data
                        success: function (data) {
                            if (data == "success") {
                                parent.layer.msg("保存成功", { icon: 1 }, function () {
                                    parent.layer.close(index);          //关闭layer.open
                                    window.parent.location.reload();   //刷新父页面
                                });
                            }
                            else {
                                parent.layer.msg("保存失败");
                            }
                        },
                        error: function (data) {
                            alert(data);
                        }
                    }
                    $.ajax(options);            //不能忘记这个
