Mongoose 增删改查 --------删

这里提供三种方案来实现 删除数据  

//第一种方案 通过manModel上的remove()来删除数据 
//Model.remove(conditions, [callback])
        
        var mongoose = require('mongoose')
        mongoose.connect('mongodb://127.0.0.1:27017/vue-admin',function(err){
            if(err){
                console.log('数据库连接失败')
            }else{
                //数据库连接成功
                var schema = new mongoose.Schema({name:String,age:Number});
                var manModel = mongoose.Model('manModel',schema);
                manModel.remove(function(err){
                   //删除所有数据
                })
            }
        })

//第二种方案 通过manModel上的deleteOne()来删除数据
//Model.deleteOne(conditions, [callback])
        
        var mongoose = require('mongoose')
        mongoose.connect('mongodb://127.0.0.1:27017/vue-admin',function(err){
            if(err){
                console.log('数据库连接失败')
            }else{
                //数据库连接成功
                var schema = new mongoose.Schema({name:String,age:Number});
                var manModel = mongoose.Model('manModel',schema);
                manModel.deleteOne({name:'xmgdty'},function(err){
                   //删除符合条件的第一个数据
                })
            }
        })

//第三种方案 通过文档的remove()方法来删除数据
//doc.remove([callback])
        
        var mongoose = require('mongoose')
        mongoose.connect('mongodb://127.0.0.1:27017/vue-admin',function(err){
            if(err){
                console.log('数据库连接失败')
            }else{
                //数据库连接成功
                var schema = new mongoose.Schema({name:String,age:Number});
                var manModel = mongoose.Model('manModel',schema);
                manModel.find({name:'xmgdty'},function(err,docs){
                    docs.forEach((item,index)=>{
                        item.remove(function(err,doc){
                            //{ _id: 8571f93be6f98ec60e3dc66c, name: 'xmgdty', age:18 }
                            //{ _id: 8571f93be6f98ec60e3dc66e, name: 'xmgdty1', age:18 }
                            console.log(doc);
                        })
                    })
                })
            }
        })

 

在我们真正做项目的时候  大多数时候 是不会通过上面的方法删除数据的 因为无论是大公司还是小公司 数据对一个公司而言都是非常重要 尤其是现在这个大数据时代  数据更能凸显价值  但是我们有时候又确实希望删除一条数据 我们可以另辟蹊径 通过某个条件查询到需要删除的数据 做一个标记  例如设置一个 show:true/false 当我们不需要展示这条数据时  我们把true修改成false 这时候我们可以直接把不需要展示给用户的数据 过滤掉 （这种思想非常重要）
