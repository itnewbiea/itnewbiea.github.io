js根据key值排序

const unordered = {
  '1': 'foo',
  '2': 'bar',
  '5': 'br',
  '13': 'ar',
  '111': 'baz'
};
 
const ordered = {};
Object.keys(unordered).sort().forEach(function(key) {
  ordered[key] = unordered[key];
});
console.log(JSON.stringify(ordered));
